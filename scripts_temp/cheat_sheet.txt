import pandas as pd

df = pd.read_table('maze_fields.csv', usecols=[0,1,2,3,4], header=None)
data = []
with open('maze_fields.csv') as inf:
    for line in inf:
        line = line.split()
        lastcolumn = line[5:]
        lastcolumn = ' '.join(lastcolumn)
        data.append(lastcolumn)

df['desc'] = data
